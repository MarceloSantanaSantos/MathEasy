CREATE TABLE professor (idProf INT (5) NOT NULL PRIMARY KEY AUTO_INCREMENT, nomeProf VARCHAR (100) NOT NULL, emailProf VARCHAR (100) NOT NULL, senhaProf VARCHAR (50) NOT NULL);

CREATE TABLE aluno (idAluno INT (5) NOT NULL PRIMARY KEY AUTO_INCREMENT, nomeAluno VARCHAR (100) NOT NULL, emailAluno VARCHAR (100) NOT NULL, senhaAluno VARCHAR (50) NOT NULL);

CREATE TABLE escola (idEScola INT (5) NOT NULL PRIMARY KEY AUTO_INCREMENT, nomeEscola VARCHAR (100) NOT NULL, cidadeEscola VARCHAR (50) NOT NULL);
